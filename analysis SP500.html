import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates

# Load and clean the data
file_path = r'C:\Users\eterra\Desktop\pt\anaconda\S&P 500 Historical Data.csv'
df = pd.read_csv(file_path, delimiter=';')
df.columns = [col.strip() for col in df.columns]

# Clean numeric columns
for col in df.columns[1:]:
    df[col] = df[col].str.replace(',', '').str.replace(' ', '').str.replace('%', '')
    df[col] = pd.to_numeric(df[col], errors='coerce')

# Convert dates
df['Date'] = pd.to_datetime(df['Date'], format='%d/%m/%Y')
df = df.sort_values('Date')

# Calculate financial metrics
df['Daily Return'] = df['Price'].pct_change()
df['MA20'] = df['Price'].rolling(window=20).mean()
df['MA50'] = df['Price'].rolling(window=50).mean()
df['Rolling Volatility (30d)'] = df['Daily Return'].rolling(window=30).std()

# Set plotting style
plt.style.use('ggplot')
font = {'size': 12}
plt.rc('font', **font)
locator = mdates.YearLocator()
formatter = mdates.DateFormatter('%Y')

# GRAPH 1: Distribution of Daily Returns
# -------------------------------
## This histogram shows the frequency of daily returns.
## Formula used: Daily Return = (Price_t / Price_{t-1}) - 1
## It gives insight into the risk/return profile, identifying the spread of gains/losses.

plt.hist(df['Daily Return'].dropna(), bins=50, edgecolor='black', color='#1f77b4')

plt.xlabel('Daily Return')
plt.ylabel('Frequency')
plt.grid(True, linestyle='--', alpha=0.6)
plt.tight_layout()
plt.show()

plt.title('S&P 500 Price with 20 & 50-Day Moving Averages', fontsize=16, fontweight='bold')

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates

# -------------------------------
# LOAD & CLEAN DATA
# -------------------------------
file_path = r'C:\Users\eterra\Desktop\pt\anaconda\S&P 500 Historical Data.csv'
df = pd.read_csv(file_path, delimiter=';')
df.columns = [col.strip() for col in df.columns]

# Clean numeric columns
for col in df.columns[1:]:
    df[col] = df[col].str.replace(',', '').str.replace(' ', '').str.replace('%', '')
    df[col] = pd.to_numeric(df[col], errors='coerce')

df['Date'] = pd.to_datetime(df['Date'], format='%d/%m/%Y')
df = df.sort_values('Date')

# -------------------------------
# CALCULATIONS
# -------------------------------
df['Daily Return'] = df['Price'].pct_change()
df['Cumulative Return'] = (1 + df['Daily Return']).cumprod()
df['Cumulative Max'] = df['Cumulative Return'].cummax()
df['Drawdown'] = (df['Cumulative Return'] - df['Cumulative Max']) / df['Cumulative Max']

# -------------------------------
# PLOT 1: Cumulative Return
## This graph shows the growth of a hypothetical $1 investment over time.
## It's calculated by compounding daily returns: (1 + return_1) * (1 + return_2) * ...
## It helps visualize the long-term performance of the S&P 500.
# -------------------------------

plt.figure(figsize=(16, 6))  # Wider figure
plt.plot(df['Date'], df['Cumulative Return'], color='#2ca02c', linewidth=2)
plt.title('Cumulative Return of S&P 500', fontsize=18, fontweight='bold')
plt.xlabel('Date', fontsize=12)
plt.ylabel('Growth of $1', fontsize=12)
plt.grid(True, linestyle='--', alpha=0.7)

# Improved x-axis formatting
plt.gca().xaxis.set_major_locator(mdates.YearLocator(2))  # every 2 years
plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%Y'))
plt.xticks(rotation=45)  # rotate for readability

plt.tight_layout()
plt.show()
# -------------------------------
# PLOT 2: Drawdown (Risk)
## This graph shows drawdowns: how far the index has fallen from previous peaks.
## It's a key risk measure in finance, capturing worst-case declines during crashes.
## Calculated as: (current value - max value so far) / max value so far.
# -------------------------------
plt.figure(figsize=(16, 6))  # Wider figure
plt.fill_between(df['Date'], df['Drawdown'], color='firebrick', alpha=0.4)
plt.plot(df['Date'], df['Drawdown'], color='firebrick', linewidth=1.5)
plt.title('Drawdown from Historical Peaks', fontsize=18, fontweight='bold')
plt.xlabel('Date', fontsize=12)
plt.ylabel('Drawdown (%)', fontsize=12)
plt.grid(True, linestyle='--', alpha=0.7)

# X-axis formatting for better readability
plt.gca().xaxis.set_major_locator(mdates.YearLocator(2))  # one tick every 2 years
plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%Y'))
plt.xticks(rotation=45)  # rotate date labels

plt.tight_layout()
plt.show()

# -------------------------------
# TABLE: Yearly Performance Summary
## This summary shows annual returns and volatility of the S&P 500.
## Volatility is the annualized standard deviation of daily returns.
## This table is useful to assess performance and risk by year.
# -------------------------------
df['Year'] = df['Date'].dt.year
yearly = df.groupby('Year').agg(
    Start_Price=('Price', 'first'),
    End_Price=('Price', 'last'),
    Mean_Return=('Daily Return', 'mean'),
    Volatility=('Daily Return', 'std')
)
yearly['Yearly Return (%)'] = (yearly['End_Price'] / yearly['Start_Price'] - 1) * 100
yearly['Annualized Volatility (%)'] = yearly['Volatility'] * (252**0.5) * 100
yearly = yearly[['Yearly Return (%)', 'Annualized Volatility (%)']].round(2)

# Print summary table
print("\nYearly Performance Summary:")
print(yearly)

import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import numpy as np

# --------------------------------------
# ANALYSIS 1: Rolling Sharpe Ratio (30-Day)
# --------------------------------------
## Formula:
##    Sharpe_t = [mean(Daily_Returns_t-29:t) - risk_free_rate] / std(Daily_Returns_t-29:t)
## We assume risk-free rate  0 for simplification.
## Insight: Measures return-to-risk ratio dynamically over time.

df['Rolling Mean Return'] = df['Daily Return'].rolling(window=30).mean()
df['Rolling Volatility'] = df['Daily Return'].rolling(window=30).std()
df['Rolling Sharpe Ratio'] = df['Rolling Mean Return'] / df['Rolling Volatility']

plt.figure(figsize=(14, 6))
plt.plot(df['Date'], df['Rolling Sharpe Ratio'], color='darkblue', linewidth=2)
plt.title('30-Day Rolling Sharpe Ratio of S&P 500', fontsize=16, fontweight='bold')
plt.xlabel('Date')
plt.ylabel('Sharpe Ratio')
plt.grid(True, linestyle='--', alpha=0.6)
plt.gca().xaxis.set_major_locator(mdates.YearLocator(2))
plt.gca().xaxis.set_major_formatter(mdates.DateFormatter('%Y'))
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# --------------------------------------
# ANALYSIS 2: Histogram of Daily Volatility (Cluster Detection)
# --------------------------------------
## Formula:
##    Volatility = std(Daily_Return)
## Insight: Shows clustering of volatility (fat tails, skewness)

volatility_values = df['Daily Return'].rolling(window=5).std()

plt.figure(figsize=(12, 6))
plt.hist(volatility_values.dropna(), bins=50, color='orange', edgecolor='black')
plt.title('Histogram of 5-Day Rolling Volatility', fontsize=16, fontweight='bold')
plt.xlabel('5-Day Std Dev of Returns')
plt.ylabel('Frequency')
plt.grid(True, linestyle='--', alpha=0.6)
plt.tight_layout()
plt.show()

# --------------------------------------
# ANALYSIS 3: Autocorrelation of Daily Returns
# --------------------------------------
## Formula:
##    (k) = Corr(Return_t, Return_{t-k})
## Insight: Detects if returns are serially correlated (momentum or mean-reversion)

from pandas.plotting import autocorrelation_plot

plt.figure(figsize=(10, 5))
autocorrelation_plot(df['Daily Return'].dropna())
plt.title('Autocorrelation of Daily Returns', fontsize=16, fontweight='bold')
plt.xlabel('Lag (Days)')
plt.grid(True, linestyle='--', alpha=0.6)
plt.tight_layout()
plt.show()
